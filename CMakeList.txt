cmake_minimum_required(VERSION 2.8)

project(symb-lib)
set(CMAKE_CXX_STANDARD 17)
set(CMAKE_CXX_STANDARD_REQUIRED ON)

set ( CMAKE_C_FLAGS "-Werror")
include_directories(symb_lib)

file(GLOB SOURCES
    symb_lib/manager/*.cpp
    symb_lib/node/*.cpp
    processor/*.cpp
    parser/*.cpp
)

file(GLOB HEADERS
    manager/*.h
    node/*.h
    processor/*.h
    parser/*.h
    Types.h
)

add_library(${PROJECT_NAME} SHARED ${HEADERS} ${SOURCES})
add_executable(run exe/main.cpp)
add_executable(example exe/example.cpp)
target_link_libraries(example ${PROJECT_NAME})
target_link_libraries(run ${PROJECT_NAME})
